    function toggleAccordion(accordion) {
        var detailsElements = document.getElementsByTagName("details");
        for (var i = 0; i < detailsElements.length; i++) {
            if (detailsElements[i] !== accordion) {
                detailsElements[i].removeAttribute("open");
            }
        }
    }
